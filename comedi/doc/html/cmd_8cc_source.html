00214         ret = comedi_get_cmd_generic_timed(dev, subdevice, cmd, n_chan, scan_period_nanosec);

00228 /*
00229  * Set up a command by hand.  This will not work on some devices.
00230  * There is no single command that will work on all devices.
00231  */
00232 int prepare_cmd(comedi_t *dev, int subdevice, int n_scan, int n_chan, unsigned period_nanosec, comedi_cmd *cmd)
00233 {
00234         memset(cmd,0,sizeof(*cmd));
00235 
00236         /* the subdevice that the command is sent to */
00237         cmd->subdev =   subdevice;
00238 
00239         /* flags */
00240         cmd->flags = 0;
00241 
00242         /* Wake up at the end of every scan */
00243         //cmd->flags |= TRIG_WAKE_EOS;
00244 
00245         /* Use a real-time interrupt, if available */
00246         //cmd->flags |= TRIG_RT;
00247 
00248         /* each event requires a trigger, which is specified
00249            by a source and an argument.  For example, to specify
00250            an external digital line 3 as a source, you would use
00251            src=TRIG_EXT and arg=3. */
00252 
00253         /* The start of acquisition is controlled by start_src.
00254          * TRIG_NOW:     The start_src event occurs start_arg nanoseconds
00255          *               after comedi_command() is called.  Currently,
00256          *               only start_arg=0 is supported.
00257          * TRIG_FOLLOW:  (For an output device.)  The start_src event occurs
00258          *               when data is written to the buffer.
00259          * TRIG_EXT:     start event occurs when an external trigger
00260          *               signal occurs, e.g., a rising edge of a digital
00261          *               line.  start_arg chooses the particular digital
00262          *               line.
00263          * TRIG_INT:     start event occurs on a Comedi internal signal,
00264          *               which is typically caused by an INSN_TRIG
00265          *               instruction.
00266          */
00267         cmd->start_src =        TRIG_NOW;
00268         cmd->start_arg =        0;
00269 
00270         /* The timing of the beginning of each scan is controlled by
00271          * scan_begin.
00272          * TRIG_TIMER:   scan_begin events occur periodically.
00273          *               The time between scan_begin events is
00274          *               convert_arg nanoseconds.
00275          * TRIG_EXT:     scan_begin events occur when an external trigger
00276          *               signal occurs, e.g., a rising edge of a digital
00277          *               line.  scan_begin_arg chooses the particular digital
00278          *               line.
00279          * TRIG_FOLLOW:  scan_begin events occur immediately after a scan_end
00280          *               event occurs.
00281          * The scan_begin_arg that we use here may not be supported exactly
00282          * by the device, but it will be adjusted to the nearest supported
00283          * value by comedi_command_test(). */
00284         cmd->scan_begin_src =   TRIG_TIMER;
00285         cmd->scan_begin_arg = period_nanosec;           /* in ns */
00286 
00287         /* The timing between each sample in a scan is controlled by convert.
00288          * TRIG_TIMER:   Conversion events occur periodically.
00289          *               The time between convert events is
00290          *               convert_arg nanoseconds.
00291          * TRIG_EXT:     Conversion events occur when an external trigger
00292          *               signal occurs, e.g., a rising edge of a digital
00293          *               line.  convert_arg chooses the particular digital
00294          *               line.
00295          * TRIG_NOW:     All conversion events in a scan occur simultaneously.
00296          * Even though it is invalid, we specify 1 ns here.  It will be
00297          * adjusted later to a valid value by comedi_command_test() */
00298         cmd->convert_src =      TRIG_TIMER;
00299         cmd->convert_arg =      1;              /* in ns */
00300 
00301         /* The end of each scan is almost always specified using
00302          * TRIG_COUNT, with the argument being the same as the
00303          * number of channels in the chanlist.  You could probably
00304          * find a device that allows something else, but it would
00305          * be strange. */
00306         cmd->scan_end_src =     TRIG_COUNT;
00307         cmd->scan_end_arg =     n_chan;         /* number of channels */
00308 
00309         /* The end of acquisition is controlled by stop_src and
00310          * stop_arg.
00311          * TRIG_COUNT:  stop acquisition after stop_arg scans.
00312          * TRIG_NONE:   continuous acquisition, until stopped using
00313          *              comedi_cancel()
00314          * */
00315         cmd->stop_src =         TRIG_COUNT;
00316         cmd->stop_arg =         n_scan;
00317 
00318         /* the channel list determined which channels are sampled.
00319            In general, chanlist_len is the same as scan_end_arg.  Most
00320            boards require this.  */
00321         cmd->chanlist =         chanlist;
00322         cmd->chanlist_len =     n_chan;
00323 
00324         return 0;
00325 }
